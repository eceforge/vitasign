609%DUAL THRESHOLD PROCESSSING
610% Filters out R_peaks which don't meet the threshold reqs
611    function [indices, noise_lvl, signal_lvl] = dualThreshold(R_peak_vals, threshold, indices, max_voltage, pos_deviance_threshold, neg_deviance_threshold)
612        Fixed_Point_Properties = numerictype('WordLength', 32, 'FractionLength', 16, 'Signed',false);
613        F = fimath('OverflowMode','saturate', 'RoundMode', 'nearest', 'ProductFractionLength', 16,'ProductMode', 'SpecifyPrecision', 'MaxProductWordLength', 32, 'SumFractionLength', 16, 'SumMode', 'SpecifyPrecision','MaxSumWordLength', 32);
614        % Asserts that the input parameters are of fixed point
615        assert(isfi(R_peak_vals));
616        assert(isfi(threshold));
617        assert(isa(indices,'int32'));
618        assert(isfi(max_voltage));
619        assert(isfi(pos_deviance_threshold));
620        assert(isfi(neg_deviance_threshold));
621        % Asserts that input parameters are of specific fixed point parameters
622        assert(isequal(numerictype(R_peak_vals), Fixed_Point_Properties) && isequal(fimath(R_peak_vals), F));
623        assert(isequal(numerictype(threshold), Fixed_Point_Properties) && isequal(fimath(threshold), F));
624        assert(isequal(numerictype(max_voltage), Fixed_Point_Properties) && isequal(fimath(max_voltage), F));
625        assert(isequal(numerictype(pos_deviance_threshold), Fixed_Point_Properties) && isequal(fimath(pos_deviance_threshold), F));
626        assert(isequal(numerictype(neg_deviance_threshold), Fixed_Point_Properties) && isequal(fimath(neg_deviance_threshold), F));
627        noise_sum = fi(0, Fixed_Point_Properties, F); signal_sum = fi(0, Fixed_Point_Properties, F);
628        noise_count = fi(0, Fixed_Point_Properties, F); signal_count = fi(0, Fixed_Point_Properties, F);
629        noise_lvl = fi(0, Fixed_Point_Properties, F); signal_lvl = fi(0,Fixed_Point_Properties, F);
630        for index=1:length(R_peak_vals)
631               % DELETE AFTER DEBUGGING
632               %if (shouldPlot && channel == 2)
633               %    fprintf('The peak val is: %f\n',R_peak_vals(index));
634               %end
635           if (R_peak_vals(index) * max_voltage > threshold) 
636               %fprintf('mv: %f. Threshold is: %f\n', R_peak_vals(index) * max_voltage, threshold);
637               %R_peak_vals(index)
638               % Filters out any signal value which exceeds the allowed deviance from
639               % the average signal value 
640               if (~meets_deviance_threshold(R_peak_vals(index), signal_lvl, pos_deviance_threshold, neg_deviance_threshold))
641                  %fprintf('%f, %f Does not meet the deviance threshold\n', R_peak_vals(index), signal_lvl);
642                   % Sets all the indices which R_vals don't meet the threshold to 0
643                   indices(index) = 0; 
644                   % Updates the average noise signal lvl
645                   noise_sum = noise_sum  + R_peak_vals(index);
646                   noise_count = noise_count + 1;
647                   % Calculates the noise level
648%                    noise_lvl = noise_sum / noise_count;
649                   noise_lvl = divide(Fixed_Point_Properties, noise_sum, noise_count);
650                   continue;
651               end
652               % DELETE AFTER DEBUGGING
653               %if (shouldPlot && channel == 2)
654               %   fprintf('The peak val is: %f\n',R_peak_vals(index));
655               %end
656                % Updates the average signal lvl
657               signal_sum = signal_sum + R_peak_vals(index);
658               signal_count = signal_count + 1;
659               % Calculates the signal level
660%                signal_lvl = signal_sum / signal_count;
661               signal_lvl = divide(Fixed_Point_Properties, signal_sum, signal_sum);
662           else
663               % Sets all the indices which R_vals don't meet the threshold to 0
664               indices(index) = 0; 
665               % Updates the average noise signal lvl
666               noise_sum = noise_sum  + R_peak_vals(index);
667               noise_count = noise_count + 1;
668               % Calculates the noise level
669%                noise_lvl = noise_sum / noise_count;
670               noise_lvl = divide(Fixed_Point_Properties, noise_sum, noise_count);
671
672           end          
673        end
674    end
