  556   static void uMultiWordShrNear(const uint32_T u1[], int32_T n1, uint32_T n2,
  557     uint32_T y[], int32_T n)
  558   {
  559     uint32_T n2m1;
  560     int32_T nb;
  561     boolean_T doNearest = FALSE;
  562     n2m1 = n2 - 1U;
  563     nb = (int32_T)n2m1 / 32;
  564     if (nb < n1) {
  565       n2m1 -= (uint32_T)(nb * 32);
  566       n2m1 = 1U << n2m1;
  567       doNearest = ((u1[nb] & n2m1) != 0U);
  568     }
  569   
  570     uMultiWordShr(u1, n1, n2, y, n);
  571     nb = 0;
  572     while (doNearest && (nb < n)) {
  573       y[nb]++;
  574       if (y[nb] != 0U) {
  575         doNearest = FALSE;
  576       }
  577   
  578       nb++;
  579     }
  580   }
