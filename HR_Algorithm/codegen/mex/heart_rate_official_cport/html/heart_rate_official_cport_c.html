  461   static void mul_wide_u32(uint32_T in0, uint32_T in1, uint32_T *ptrOutBitsHi, uint32_T *ptrOutBitsLo)
  462   {
  463       int32_T in0Hi;
  464       int32_T in0Lo;
  465       int32_T in1Hi;
  466       int32_T in1Lo;
  467       uint32_T productHiHi;
  468       uint32_T productHiLo;
  469       uint32_T productLoHi;
  470       uint32_T productLoLo;
  471       uint32_T outBitsLo;
  472       in0Hi = (int32_T)(in0 >> 16U);
  473       in0Lo = (int32_T)(in0 & 65535U);
  474       in1Hi = (int32_T)(in1 >> 16U);
  475       in1Lo = (int32_T)(in1 & 65535U);
  476       productHiHi = (uint32_T)in0Hi * (uint32_T)in1Hi;
  477       productHiLo = (uint32_T)in0Hi * (uint32_T)in1Lo;
  478       productLoHi = (uint32_T)in0Lo * (uint32_T)in1Hi;
  479       productLoLo = (uint32_T)in0Lo * (uint32_T)in1Lo;
  480       in0Hi = 0;
  481       outBitsLo = productLoLo + (productLoHi << 16U);
  482       if (outBitsLo < productLoLo) {
  483           in0Hi = 1;
  484       }
  485       productLoLo = outBitsLo;
  486       outBitsLo += productHiLo << 16U;
  487       if (outBitsLo < productLoLo) {
  488           in0Hi = (int32_T)((uint32_T)in0Hi + 1U);
  489       }
  490       productHiHi = (((uint32_T)in0Hi + productHiHi) + (productLoHi >> 16U)) + (productHiLo >> 16U);
  491       *ptrOutBitsHi = productHiHi;
  492       *ptrOutBitsLo = outBitsLo;
  493   }
